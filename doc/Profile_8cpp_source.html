69double Profile::getDistance(const Profile &otherProfile) const {
70 if (this->_vectorKmerFreq.empty() || otherProfile._vectorKmerFreq.empty())
71 throw ::std::invalid_argument(
72 "Every Profile must have at least one KmerFreq");
73 double distance = 0;
74 int rank;
75
76 for (int i = 0; i < getSize(); i++) {
77 rank = otherProfile.findKmer(at(i).getKmer());
78 if (rank == -1)
79 rank = otherProfile.getSize();
80 distance = distance + abs(i - rank);
81 }
82
83 distance = distance / (getSize() * otherProfile.getSize());
84 return distance;
85}
