60Kmer Kmer::complementary(const string &nucleotides,
61 const string &complementaryNucleotides) const {
62 if (nucleotides.size() != complementaryNucleotides.size())
63 throw std::invalid_argument(
64 "Error: \"nucleotides and \"complementaryNucleotides\" must have "
65 "the same length");
66 Kmer comp = *this;
67
68 for (int i = 0; i < size(); i++) {
69 if (IsValidNucleotide(at(i), nucleotides)) {
